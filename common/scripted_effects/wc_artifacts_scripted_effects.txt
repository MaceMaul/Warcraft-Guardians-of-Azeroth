set_prefix_and_suffix_flags_to_weapon_artifact_effect = {
	set_prefix_10_temp_flags_artifact_effect = yes
	if = {
		limit = {
			NOR = {
				has_artifact_flag = artifact_weapon_element_fire_flag
				has_artifact_flag = artifact_weapon_element_frost_flag
				has_artifact_flag = artifact_weapon_element_lightning_flag
				has_artifact_flag = artifact_weapon_element_poison_flag
				has_artifact_flag = artifact_weapon_element_magic_flag

give_generic_artifact_random_name_and_origin_effect = {
	if = {
		limit = { quality == 1 }
		set_description = "artifact_common_origin"
	}
	else_if = {
		limit = { quality == 2 }
		set_description = "artifact_uncommon_origin"
	}
	else_if = {
		limit = { quality == 3 }
		set_description = "artifact_rare_origin"
	}
	else_if = {
		limit = { quality == 4 }
		set_description = "artifact_epic_origin"
	}
	else = { set_description = "artifact_legendary_origin" }
	give_generic_artifact_random_name_effect = yes
}

finish_artifact_creation_effect = {
	set_creation_date = 1.1.1
	log = "Added generic artifact: [This.GetName]"
}
